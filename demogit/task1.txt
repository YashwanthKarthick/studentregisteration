Use Case Scenario: You are tasked with creating a simple task management application that allows users to perform basic operations on tasks. The application should have the following features:
1. Task Creation:
 * Users can create new tasks by providing a title and a description.
 * Each task should have a unique identifier.
2. Task Listing:
 * Users can view a list of all the tasks.
 * The list should display the task titles along with their unique identifiers.
3. Task Completion:
 * Users can mark a task as completed.
 * Completed tasks should be visually differentiated from pending tasks in the task list.
4. Task Deletion:
 * Users can delete a task from the list.
 * Once deleted, the task should be removed from the task list.
5. User Interface:
 * Design a simple command-line interface (CLI) for the application.
 * Display clear instructions and user prompts to guide interactions.



import java.util.*;

public class Main {
    public static void select() {
        System.out.println("Enter the choices to be done?\n1.Add\n2.View\n3.Delete\n4.Exit");

    }
    public static void main(String[] args) {
        String a;
        Scanner sc;
        Scanner s1=new Scanner(System.in);
        int choice;
        List<String> list = new ArrayList<>();
        while (true) {
            sc = new Scanner(System.in);
            select();
            choice = sc.nextInt();
            switch (choice) {
                case 1:
                    System.out.println("ENTER THE NO OF TASKS");
                    int no = sc.nextInt();
                    System.out.println("ENTER THE LIST OF TASKS");
                    for (int i = 0; i < no; i++) {
                        list.add(s1.nextLine());
                    }
                    break;
                case 2:
                    System.out.print("THE GIVEN LIST IS\n");
                    for(String item:list)
                        System.out.println(item);
                    break;
                case 3:
                    System.out.println("Enter index which one you need to delete");
                    int index=sc.nextInt();
                    sc.nextLine();
                    if(index>=0 && index < list.size()){
                        String dllist=list.remove(index);
                        System.out.println("the deleted name is "+dllist);
                    }
                    else{
                        System.out.println("invalid index");
                    }
                    break;
            }
                if(choice==4){
                    System.out.println("Thank you");
                    break;
                }
            }
        }

    }
